import numpy as np
def QR_Decomposition(a):
    n,m=a.shape
    q=np.empty((n,m))
    u=np.empty((n,m))
    r=np.zeros((n,m))
    u[:,0]=a[:,0]
    q[:,0]=u[:,0]/np.linalg.norm(u[:,0])
    for i in range(1,n):
        u[:,i]=a[:,i]
        for j in range(n):
            u[:,i]-=(a[:,i]@q[:,j])*q[:,j]
        q[:,i]=u[:,i]/np.linalg.norm(u[:,i])
    for i in range(n):
        for j in range(i,m):
            r[i,j]=a[:,j]@q[:,i]
    print(f"The Q Matrix is\n {q}")
    print(f"The R Matrix is\n {r}")

a = np.array(eval(input()))
QR_Decomposition(a)
